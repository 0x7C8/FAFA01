\begin{equation} \label{eq:I_t}
	I(t) = I_0e^{-t/\tau}
\end{equation}

\begin{equation} \label{eq:log_I}
	ln(I(t)) = -\dfrac{t}{\tau} + ln(I_0)
\end{equation}
